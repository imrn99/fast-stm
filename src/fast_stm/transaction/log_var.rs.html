1use std::any::Any;
2use std::sync::Arc;
3
4pub type ArcAny = Arc<dyn Any + Send + Sync>;
5
6/// `LogVar` is used by `Log` to track which `Var` was either read or written or both.
7/// Depending on the type, STM has to write, ensure consistency or block on this value.
8#[derive(Clone)]
9pub enum LogVar {
10    /// Var has been read.
11    Read(ArcAny),
12
13    /// Var has been written and no dependency on the original exists.
14    ///
15    /// There is no need to check for consistency.
16    Write(ArcAny),
17
18    /// ReadWrite(original value, temporary stored value).
19    ///
20    /// Var has been read first and then written.
21    ///
22    /// It needs to be checked for consistency.
23    ReadWrite(ArcAny, ArcAny),
24
25    /// Var has been read on blocked path.
26    ///
27    /// Don't check for consistency, but block on Var,
28    /// so that the threat wakes up when the first path
29    /// has been unlocked.
30    ReadObsolete(ArcAny),
31
32    /// ReadWriteObsolete(original value, temporary stored value).
33    ///
34    /// Var has been read on blocked path and then written to.
35    ///
36    /// Don't check for consistency, but block on Var,
37    /// so that the threat wakes up when the first path
38    /// has been unlocked.
39    ReadObsoleteWrite(ArcAny, ArcAny), // Here would be WriteObsolete, but the write onlies can be discarded immediately
40                                       // and don't need a representation in the log.
41}
42
43impl LogVar {
44    /// Read a value and potentially upgrade the state.
45    pub fn read(&mut self) -> ArcAny {
46        // We do some kind of dance around the borrow checker here.
47        // Ideally we only clone the read value and not the write,
48        // in order to avoid hitting shared memory as least as possible,
49        // but we can not fully avoid it, although these cases happen rarely.
50        let this;
51        let val;
52        match &*self {
53            // Use last read value or get written one
54            &Self::Read(ref v) | &Self::Write(ref v) | &Self::ReadWrite(_, ref v) => {
55                return v.clone();
56            }
57
58            Self::ReadObsoleteWrite(w, v) => {
59                val = v.clone();
60                this = Self::ReadWrite(w.clone(), v.clone());
61            }
62
63            // Upgrade to a real Read
64            Self::ReadObsolete(v) => {
65                val = v.clone();
66                this = Self::Read(v.clone());
67            }
68        }
69        *self = this;
70        val
71    }
72
73    /// Write a value and potentially upgrade the state.
74    pub fn write(&mut self, w: ArcAny) {
75        let this = self.clone();
76
77        *self = match this {
78            Self::Write(_) => Self::Write(w),
79
80            // Register write
81            Self::ReadObsolete(r) | Self::ReadObsoleteWrite(r, _) => Self::ReadObsoleteWrite(r, w),
82
83            // Register write
84            Self::Read(r) | Self::ReadWrite(r, _) => Self::ReadWrite(r, w),
85        };
86    }
87
88    /// Turn `self` into an obsolete version.
89    pub fn obsolete(self) -> Option<LogVar> {
90        self.into_read_value().map(LogVar::ReadObsolete)
91    }
92
93    /// Ignore all Write... and get the original value of a Var.
94    pub fn into_read_value(self) -> Option<ArcAny> {
95        match self {
96            LogVar::Read(v)
97            | LogVar::ReadWrite(v, _)
98            | LogVar::ReadObsolete(v)
99            | LogVar::ReadObsoleteWrite(v, _) => Some(v),
100            LogVar::Write(_) => None,
101        }
102    }
103}
